## Returns

<ResponseField name="id" type="string">
  Response ID (format: `resp_` + 48 hex chars).
</ResponseField>

<ResponseField name="object" type="string">
  Always "response".
</ResponseField>

<ResponseField name="created_at" type="integer">
  Unix timestamp (seconds).
</ResponseField>

<ResponseField name="status" type="string">
  "completed", "failed", "incomplete", "in_progress", "cancelled", "queued".
</ResponseField>

<ResponseField name="error" type="object/null">
  Error details if `status` is "failed".
</ResponseField>

<ResponseField name="incomplete_details" type="object/null">
  Reason if `status` is "incomplete".
</ResponseField>

<ResponseField name="instructions" type="string/array/null">
  Echoed instructions parameter.
</ResponseField>

<ResponseField name="max_output_tokens" type="integer/null">
  Echoed max output tokens.
</ResponseField>

<ResponseField name="model" type="string">
  Model used for generation.
</ResponseField>

<ResponseField name="output" type="array">
  Array of output items (messages, function calls, etc.).
</ResponseField>

<ResponseField name="parallel_tool_calls" type="boolean">
  Whether parallel tool calls are enabled.
</ResponseField>

<ResponseField name="previous_response_id" type="string/null">
  Previous response ID if continuation.
</ResponseField>

<ResponseField name="reasoning" type="object">
  Reasoning effort and summary.
</ResponseField>

<ResponseField name="store" type="boolean">
  Whether conversation was stored.
</ResponseField>

<ResponseField name="temperature" type="number">
  Temperature used.
</ResponseField>

<ResponseField name="text" type="object">
  Text format configuration.
</ResponseField>

<ResponseField name="tool_choice" type="string/object">
  Tool choice strategy.
</ResponseField>

<ResponseField name="tools" type="array">
  Available tools.
</ResponseField>

<ResponseField name="top_p" type="number">
  Nucleus sampling parameter.
</ResponseField>

<ResponseField name="truncation" type="string">
  Truncation strategy.
</ResponseField>

<ResponseField name="usage" type="object">
  Detailed token usage.
  
  <Expandable title="usage properties">
    <ResponseField name="input_tokens" type="integer">
      Number of input tokens.
    </ResponseField>

    <ResponseField name="input_tokens_details" type="object">
      Cached, text, and audio token breakdown.
    </ResponseField>
    
    <ResponseField name="output_tokens" type="integer">
      Number of output tokens.
    </ResponseField>
    
    <ResponseField name="output_tokens_details" type="object">
      Reasoning, text, and audio token breakdown.
    </ResponseField>
    
    <ResponseField name="total_tokens" type="integer">
      Total tokens used.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="user" type="string/null">
  User identifier if provided.
</ResponseField>

<ResponseField name="metadata" type="object">
  Custom metadata if provided.
</ResponseField>
